#include<stdio.h>
#include<conio.h>
void bubble_sort(int a[],int n)
{
 int pass,temp, j;
for(pass=1;pass<n;pass++)
{
 for(j=0;j<=n-pass-1:j++)
 {
  if([j]>a[j+1])
   {  
     temp=a[j];
     a[j]=a[j+1];
     a[j+1]=temp;
    }
  }
 }
}
void main()
{ 
   int i,a[20],n,;
   printf("Enter the number of elements:");
   scanf("%d",&n);
   printf("Enter the array elements:");
   for(i=0;i<n;i++)
        scanf("%d",&a[i]);
   bubble_sort(a,n);
   printf("\nThe sorted elements are:\n);
   for(i=0;i<n;i++)
       printf("%d",a[i]);
   getch();
}
